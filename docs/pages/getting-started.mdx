import { useEffect, useRef } from 'react'

export function CurrentVersion({ children }) {
  const ref = useRef()
  // Find the corresponding token from the DOM
  useEffect(() => {
    if (ref.current) {
      // Jolt Version
      const joltVersionEl = [...ref.current.querySelectorAll('code > span > span')]
              .find(el => el.innerText === 'joltVersion');
      if (joltVersionEl) {
        joltVersionEl.innerText = `^${joltVersion}`;
      }
      // SDK Version
      const sdkVersionEl = [...ref.current.querySelectorAll('code > span > span')]
              .find(el => el.innerText === 'sdkVersion');
      if (sdkVersionEl) {
        sdkVersionEl.innerText = `"${dartVersion}"`;
      }
      // Flutter Version
      const flutterVersionEl = [...ref.current.querySelectorAll('code > span > span')]
              .find(el => el.innerText === 'flutterVersion');
      if (flutterVersionEl) {
        flutterVersionEl.innerText = `"${flutterVersion}"`;
      }
    }
  }, [])
  return (
    <>
      <div ref={ref} style={{ marginTop: '1.5rem' }}>
        {children}
      </div>
    </>
  )
}
